# Metabase deployment with no MB_ENCRYPTION_SECRET_KEY anywhere.
# Application DB will store source credentials in cleartext.

    image: metabase/metabase:v0.49.6
    ports:
      - "3000:3000"
    environment:
      MB_DB_TYPE: postgres
      MB_DB_HOST: pg
      MB_DB_PORT: 5432
      MB_DB_USER: metabase
      MB_DB_DBNAME: metabase
      MB_DB_PASS: hunter2
      MB_SITE_URL: https://bi.example.com/
