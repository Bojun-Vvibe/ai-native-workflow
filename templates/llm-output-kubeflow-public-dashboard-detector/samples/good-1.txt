apiVersion: security.istio.io/v1
kind: AuthorizationPolicy
metadata:
  name: centraldashboard-require-auth
  namespace: kubeflow
spec:
  selector:
    matchLabels:
      app.kubernetes.io/name: centraldashboard
  action: ALLOW
  rules:
    - from:
        - source:
            requestPrincipals: ["*"]
      when:
        - key: request.auth.claims[email]
          values: ["*"]
