# Looks safe at a glance: basic auth is "active". But the password
# field was templated from an env var that was never populated, so
# n8n boots with an empty password and accepts any credential pair.
N8N_HOST=n8n.example.net
N8N_BASIC_AUTH_ACTIVE=true
N8N_BASIC_AUTH_USER=admin
N8N_BASIC_AUTH_PASSWORD=
WEBHOOK_URL=https://n8n.example.net/
